# snouty help and version

# version subcommand prints version string.
snouty version
stdout 'snouty \d+\.\d+\.\d+'

# --version flag prints version string.
snouty --version
stdout 'snouty \d+\.\d+\.\d+'

# --help shows primary subcommands.
snouty --help
stdout '.*run.*'
stdout '.*debug.*'
stdout '.*version.*'
stdout '.*api.*'
stdout '.*docs.*'
stdout '.*completions.*'
stdout '.*update.*'
stdout '.*validate.*'

# api --help shows webhook subcommand.
snouty api --help
stdout '.*webhook.*'
