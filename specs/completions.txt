# snouty completions
#
# Generate shell completion scripts.

# Bash completions produce a valid completion script.
snouty completions bash
stdout '.*complete.*'
stdout '.*snouty.*'

# Zsh completions produce a valid completion script.
snouty completions zsh
stdout '.*compdef.*'
stdout '.*snouty.*'

# Fish completions produce a valid completion script.
snouty completions fish
stdout '.*complete.*'
stdout '.*snouty.*'

# Elvish completions produce a valid completion script.
snouty completions elvish
stdout '.*snouty.*'

# Unsupported shells are rejected.
! snouty completions nushell
stderr 'unsupported shell: nushell.*'
