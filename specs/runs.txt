# snouty runs
#
# List all Antithesis runs.

# The command authenticates with the Antithesis API using
#    ANTITHESIS_USERNAME, ANTITHESIS_PASSWORD, and ANTITHESIS_TENANT env vars.
! snouty runs
stderr 'missing environment variable.*'

# On success, the command fetches all pages via next_cursor and prints a table.
mock-runs-server
snouty runs
stdout 'RUN ID.*STATUS.*TYPE.*CREATED AT.*LAUNCHER'
stdout 'run-1.*completed.*test.*2025-03-20T02:00:00\+00:00.*nightly'
stdout 'run-2.*in_progress.*mvd.*2025-03-19T14:00:00\+00:00.*debug'

# If no runs are returned, the command prints a friendly empty-state message.
mock-runs-server empty
snouty runs
stdout 'No runs found\.'

# API errors are reported with status code and body.
mock-server 400 '{"error":"bad_request","message":"bad request"}'
! snouty runs
stderr 'API error: 400.*'

# `snouty runs list` is the same as bare `snouty runs`.
mock-runs-server
snouty runs list
stdout 'RUN ID.*STATUS.*TYPE.*CREATED AT.*LAUNCHER'
stdout 'run-1.*completed.*test.*2025-03-20T02:00:00\+00:00.*nightly'
stdout 'run-2.*in_progress.*mvd.*2025-03-19T14:00:00\+00:00.*debug'

# `snouty runs show` displays run details in key-value format.
mock-runs-server
snouty runs show run-1
stdout 'Run ID.*run-1'
stdout 'Status.*completed'
stdout 'Launcher.*nightly'

# `snouty --json runs show` outputs JSON.
mock-runs-server
snouty --json runs show run-1
stdout '"run_id".*"run-1"'
stdout '"status".*"completed"'

# `snouty runs show` requires a run ID.
! snouty runs show
stderr '.*'

# `snouty runs properties` shows failing properties by default as flattened event rows.
mock-runs-server
snouty runs properties run-1
stdout 'EXAMPLE.*MOMENT.*NAME'
stdout 'Failing.*Fa84QWiAxKy2RNvUNjFJwm.*Counter value stays below limit'
stdout 'Failing.*Fa84QWiAxKOaZBZOzDzEfI.*Counter value stays below limit'
stdout 'Passing.*Fa84QWiAxJp8gzDJaij9No.*Counter value stays below limit'

# `snouty runs properties --all` includes passing properties too.
mock-runs-server
snouty runs properties --all run-1
stdout 'Failing.*Fa84QWiAxKy2RNvUNjFJwm.*Counter value stays below limit'
stdout 'Failing.*Fa84QWiAxKOaZBZOzDzEfI.*Counter value stays below limit'
stdout 'Passing.*Fa84QWiAxJp8gzDJaij9No.*Counter value stays below limit'
stdout 'Passing.*Fa84QWiAxJFgomrCl8pPP6.*Setup completes'

# `snouty --json runs properties` outputs JSON.
mock-runs-server
snouty --json runs properties run-1
stdout '"name".*"Counter value stays below limit"'
stdout '"status".*"Failing"'

# `snouty runs properties` prints a friendly empty-state message when no properties exist.
mock-runs-server
snouty runs properties run-empty
stdout 'No properties found\.'

# `snouty runs properties` prints a friendly message when properties exist but have no sampled events.
mock-runs-server
snouty runs properties run-no-events
stdout 'No sampled property events found\.'

# `snouty runs properties` requires a run ID.
! snouty runs properties
stderr '.*'

# `snouty runs build-logs` streams formatted build log lines.
mock-runs-server
snouty runs build-logs run-1
stdout '2025-03-20T02:01:12Z \[out\] Building image payments-service\.\.\.'
stdout '2025-03-20T02:01:15Z \[err\] Warning: deprecated feature'
stdout '2025-03-20T02:01:20Z \[out\] Build complete'

# `snouty --json runs build-logs` outputs raw NDJSON.
mock-runs-server
snouty --json runs build-logs run-1
stdout '"timestamp".*"text"'

# `snouty runs logs` outputs a table by default.
mock-runs-server
snouty runs logs run-1 Fa84QWiAxKpsvhP7fglpSa
stdout 'VTIME.*SOURCE.*OUTPUT'
stdout '1\.0.*\[app:out\].*\{"level":"info","msg":"starting"\}'
stdout '2\.0.*\[app:error\].*\{"level":"warn","msg":"slow request"\}'
stdout '3\.0.*\[app:out\].*line one\\\\nline two'
stdout '73\.94233945617452.*\[control:error\].*Post "http://10\.0\.1\.123:8003/"'
stdout '311\.8487535319291.*\[control:assert\].*UNHIT AlwaysOrUnreachable "Counter'\''s value retrieved" @ control\.go:get:87'

# `snouty runs logs --begin` resumes streaming from a given vtime.
mock-runs-server
snouty runs logs run-1 Fa84QWiAxKpsvhP7fglpSa --begin 10.0
stdout 'VTIME.*SOURCE.*OUTPUT'

# `snouty runs logs --begin-moment` resumes streaming from a copy-pasteable token.
mock-runs-server
snouty runs logs run-1 Fa84QWiAxKpsvhP7fglpSa --begin-moment Fa84QWiAxIvq0ayfqGCnei
stdout 'VTIME.*SOURCE.*OUTPUT'

# `snouty --json runs logs` outputs raw NDJSON.
mock-runs-server
snouty --json runs logs run-1 Fa84QWiAxKpsvhP7fglpSa
stdout 'output_text.*starting'
stdout 'output_text.*slow request'
stdout '"antithesis_assert".*"display_type":"AlwaysOrUnreachable"'

# `snouty --json runs logs` emits a record whose output_text contains
# a JSON-escaped newline on a single output line.
mock-runs-server
snouty --json runs logs run-1 Fa84QWiAxKpsvhP7fglpSa
stdout -count=1 'line one\\nline two'

# `snouty runs logs` requires a moment token.
mock-runs-server
! snouty runs logs run-1
stderr '.*<MOMENT>.*'

# `snouty runs events` outputs a table by default.
mock-runs-server
snouty runs events run-1 request
stdout '^MOMENT\s+SOURCE\s+OUTPUT'
stdout 'Fa84QWiAxIvq0ayfqGCnei.*\[app:error\].*\{"level":"warn","msg":"slow request"\}'

# `snouty --json runs events` outputs raw NDJSON.
mock-runs-server
snouty --json runs events run-1 counter
stdout '"antithesis_assert".*"Counter'\''s value retrieved"'
stdout '.*"moment".*"input_hash":"-4735081784258020614".*'
stdout '.*"moment".*"vtime":"311\.8487535319291".*'

# `snouty runs events` prints a friendly empty-state message.
mock-runs-server
snouty runs events run-1 nomatch
stdout 'No matching events found\.'

# `snouty runs events` requires a query.
mock-runs-server
! snouty runs events run-1
stderr '.*'
