# snouty help and version

# version subcommand prints version string.
snouty version
stdout 'snouty \d+\.\d+\.\d+'

# --version flag prints version string.
snouty --version
stdout 'snouty \d+\.\d+\.\d+'

snouty --help

stdout '.*update.*'
stdout '.*validate.*'
stdout '.*doctor.*'

# api --help shows webhook subcommand.
