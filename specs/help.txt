# snouty help and version

# version subcommand prints version string.
snouty version
stdout 'snouty \d+\.\d+\.\d+'

# --version flag prints version string.
snouty --version
stdout 'snouty \d+\.\d+\.\d+'

# --help shows primary subcommands.
snouty --help
stdout '.*run.*'
stdout '.*debug.*'
stdout '.*version.*'
stdout '.*api.*'
stdout '.*docs.*'
stdout '.*completions.*'
stdout '.*update.*'
stdout '.*validate.*'

# api --help shows webhook subcommand.
snouty api --help
stdout '.*webhook.*'

# run --help explains the compose build stage.
snouty run --help
stdout '.*compose build.*'
stdout '.*--no-build.*'

# validate --help explains how setup-complete is detected.
snouty validate --help
stdout '.*compose build.*'
stdout '.*--no-build.*'
stdout '.*\/tmp\/antithesis.*'
stdout '.*ANTITHESIS_OUTPUT_DIR.*'
stdout '.*ANTITHESIS_SDK_LOCAL_OUTPUT.*'
stdout '.*JSONL.*'
stdout '.*not by scraping compose logs.*'
stdout '.*one random first_ script.*'
stdout '.*additional first_ scripts are skipped.*'
