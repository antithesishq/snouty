# snouty help and version

# 1. version subcommand prints version string.
snouty version
stdout 'snouty \d+\.\d+\.\d+'

# 2. --version flag prints version string.
snouty --version
stdout 'snouty \d+\.\d+\.\d+'

# 3. --help shows primary subcommands.
snouty --help
stdout '.*run.*'
stdout '.*debug.*'
stdout '.*version.*'
stdout '.*api.*'
stdout '.*docs.*'
stdout '.*completions.*'
stdout '.*update.*'
stdout '.*validate.*'

# 4. api --help shows webhook subcommand.
snouty api --help
stdout '.*webhook.*'
