# snouty help and version

# version subcommand prints version string.
snouty version
stdout 'snouty \d+\.\d+\.\d+'

# --version flag prints version string.
snouty --version
stdout 'snouty \d+\.\d+\.\d+'

# --help shows primary subcommands.
snouty --help
stdout '.*debug.*'
stdout '.*version.*'
stdout '.*api.*'
stdout '.*docs.*'
stdout '.*completions.*'
stdout '.*update.*'
stdout '.*validate.*'
stdout '.*doctor.*'

# api --help shows webhook subcommand.
snouty api --help
stdout '.*webhook.*'

# validate --help documents both the compose and Kubernetes flows.
snouty validate --help
stdout '.*Compose configs:.*'
stdout '.*docker-compose locally.*'
stdout '.*setup-complete.*'
stdout '.*Scripts are not executed.*'
stdout '.*Kubernetes configs:.*'
stdout '.*k8s-validator.*'
stdout '.*manifests\/.*'
