# snouty login
#
# Interactively gather configuration and credentials and persist them in the
# global Snouty settings (settings.toml) and credentials (credentials.toml)
# files.
#
# As a developer, I want to run `snouty login` and be prompted for the tenant,
# repository, and credentials I want to use so that later commands resolve them
# automatically. Values supplied as flags or sensitive values are not
# re-prompted, and a previously persisted login is offered back as the default.
#
# `login` never contacts the API, so the whole file runs in both mock and
# staging modes. Each section pins its own temporary $HOME via `isolate-home`
# so the persisted files do not leak between sections or into the developer's
# real configuration. Prompts are written to stdout; stdin feeds the answers,
# one per line, in the order the prompts appear.

# 1. A fresh login with no flags prompts for the tenant, the repository, and the
#    kind of credentials, then persists them. With nothing stored yet, no
#    "previous value" default is offered for the tenant or repository.
isolate-home apikey
stdin login_apikey.txt
snouty login
stdout 'What Antithesis tenant would you like to use\?.*'
stdout 'What container repository would you like to use\?.*'
stdout 'What kind of credentials would you like to use\?.*'
stdout '1\. Skip setup \(Select this option if you plan to use environment variables instead of persisted credentials\.\)'
stdout '2\. API key'
stdout '3\. Username/password'
stdout '\(Hit enter to use the default value of \[1\]\)'
! stdout 'Hit enter to use the previous value'
file ~/.config/snouty/settings.toml 'tenant = "mytenant"'
file ~/.config/snouty/settings.toml 'repository = "myrepo"'
file ~/.config/snouty/credentials.toml '\[default\]'
file ~/.config/snouty/credentials.toml 'type = "ApiKey"'
file ~/.config/snouty/credentials.toml 'api_key = "sk-test-key"'

# 2. A second login (same $HOME) reads the values persisted by step 1 and offers
#    them back as the default. Leaving the API key blank reuses the stored key —
#    which only succeeds because credentials.toml was written in step 1 — so the
#    command still exits successfully without a fresh secret being entered.
stdin login_reuse.txt
snouty login
stdout '\(Hit enter to use the previous value of \[mytenant\]\)'
stdout '\(Hit enter to use the previous value of \[myrepo\]\)'

# 3. Values supplied as flags are not prompted for: with --tenant and
#    --repository given, only the credential prompt is shown.
isolate-home flags
stdin login_flags.txt
snouty login --tenant mytenant --repository myrepo
stdout 'What kind of credentials would you like to use\?.*'
! stdout 'What Antithesis tenant would you like to use'
! stdout 'What container repository would you like to use'

# 4. Selecting option 3 collects a username and password instead of an API key.
#    The username is a normal prompt; the password is read without echo and is
#    not itself echoed to stdout.
isolate-home password
stdin login_password.txt
snouty login
stdout 'What kind of credentials would you like to use\?.*'
! stdout 'ppass'
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
file ~/.config/snouty/settings.toml 'repository = "prepo"'
file ~/.config/snouty/credentials.toml '\[default\]'
file ~/.config/snouty/credentials.toml 'type = "Password"'
file ~/.config/snouty/credentials.toml 'username = "puser"'
file ~/.config/snouty/credentials.toml 'password = "ppass"'

# 5. The global --profile flag scopes the persisted login to a named profile. A
#    fresh profile has no stored default; a second login under the same profile
#    offers the values persisted by the first.
isolate-home profile
stdin login_profile.txt
snouty --profile prod login
! stdout 'Hit enter to use the previous value'
stdin login_profile_reuse.txt
snouty --profile prod login
stdout '\(Hit enter to use the previous value of \[ptenant2\]\)'

# 6. The tenant becomes the host in the derived API URL, so a tenant carrying
#    URL-significant characters is rejected before anything is persisted.
isolate-home invalid
stdin login_bad_tenant.txt
! snouty login
stderr '.*invalid tenant.*'

# 7. With settings resolution broken, `--profile prod` still scopes the persisted
#    tenant and repository under `[profile.prod]` in the global settings file, and
#    login exits successfully despite the unreadable settings file.
isolate-home with_profile
env SNOUTY_SETTINGS_PATH=broken.toml
stdin login_repair_broken_toml.txt
snouty --profile prod login
file ~/.config/snouty/settings.toml '\[profile\.prod\]'
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
file ~/.config/snouty/settings.toml 'repository = "prepo"'
file ~/.config/snouty/credentials.toml '\[profile\.prod\]'
file ~/.config/snouty/credentials.toml 'type = "Password"'
file ~/.config/snouty/credentials.toml 'username = "puser"'
file ~/.config/snouty/credentials.toml 'password = "ppass"'

# 8. The same broken settings file without `--profile` persists the values at the
#    top level instead — proving it is the flag, not some value resurrected from
#    the unreadable file, that produced the `[profile.prod]` scoping in step 7.
isolate-home without_profile
env SNOUTY_SETTINGS_PATH=broken.toml
stdin login_repair_broken_toml.txt
snouty login
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
! file ~/.config/snouty/settings.toml '\[profile'
file ~/.config/snouty/credentials.toml 'username = "puser"'
! file ~/.config/snouty/credentials.toml '\[profile'

# 9. An existing settings file should be updated by `snouty login`, not overwritten
isolate-home with_existing_global_profile
env SNOUTY_SETTINGS_PATH=""
stdin login_password.txt
snouty login
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
! file ~/.config/snouty/settings.toml 'tenant = "global-tenant"'
file ~/.config/snouty/settings.toml 'repository = "prepo"'
! file ~/.config/snouty/settings.toml 'repository = "global-repository"'
file ~/.config/snouty/settings.toml 'container_engine = "docker"'
file ~/.config/snouty/settings.toml '\[profile\.prod\]'
file ~/.config/snouty/settings.toml 'tenant = "profile-tenant"'
file ~/.config/snouty/settings.toml 'repository = "profile-repository"'

# 10. An existing settings file profile should be updated by `snouty login`, not overwritten
isolate-home with_existing_profile
env SNOUTY_SETTINGS_PATH=""
stdin login_password.txt
snouty --profile prod login
file ~/.config/snouty/settings.toml 'tenant = "global-tenant"'
file ~/.config/snouty/settings.toml 'repository = "global-repository"'
file ~/.config/snouty/settings.toml 'container_engine = "docker"'
file ~/.config/snouty/settings.toml '\[profile\.prod\]'
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
! file ~/.config/snouty/settings.toml 'tenant = "old-tenant"'
file ~/.config/snouty/settings.toml 'repository = "prepo"'
! file ~/.config/snouty/settings.toml 'repository = "old-repository"'
file ~/.config/snouty/settings.toml 'container_engine = "podman"'
file ~/.config/snouty/settings.toml '\[profile\.other\]'
file ~/.config/snouty/settings.toml 'tenant = "other-tenant"'
file ~/.config/snouty/settings.toml 'repository = "other-repository"'

# 11. Exits early with invalid input
isolate-home invalid_input
stdin login_invalid_tenant.txt
! snouty login
stderr ".*a tenant must be a valid hostname.*"

# 12. An unparsable global settings file is backed up to settings.toml.bak rather
#     than silently discarded, and a fresh settings file is written in its place.
#     The invalid file makes settings resolution fail, so login first prompts to
#     proceed (answered "1" by the stdin fixture) before repairing the file.
isolate-home unparsable_settings
stdin login_repair_broken_toml.txt
snouty login
file ~/.config/snouty/settings.toml.bak 'unparsable-settings-marker'
file ~/.config/snouty/settings.toml 'tenant = "ptenant"'
! file ~/.config/snouty/settings.toml 'unparsable-settings-marker'

# 13. An unparsable credentials file is likewise backed up to credentials.toml.bak
#     before a fresh one is written. Credentials are not part of settings
#     resolution, so login does not prompt to proceed here — it just warns and
#     backs the file up while persisting the new credentials.
isolate-home unparsable_credentials
stdin login_password.txt
snouty login
file ~/.config/snouty/credentials.toml.bak 'unparsable-credentials-marker'
file ~/.config/snouty/credentials.toml 'username = "puser"'
! file ~/.config/snouty/credentials.toml 'unparsable-credentials-marker'

# 14. `--profile ""` records the login at the top level (the defaults), *regardless*
#     of ANTITHESIS_PROFILE. An explicit empty `--profile` neutralizes the env var
#     rather than falling back to it, so nothing is scoped under `[profile.<env>]`.
isolate-home empty_profile_flag
env ANTITHESIS_PROFILE=envprofile
stdin login_empty_profile.txt
snouty --profile "" login
file ~/.config/snouty/settings.toml 'tenant = "atenant"'
file ~/.config/snouty/settings.toml 'repository = "arepo"'
! file ~/.config/snouty/settings.toml '\[profile'

# 15. Control for step 14: with the same ANTITHESIS_PROFILE set but *no* `--profile`
#     flag, the env var is honored and the login is scoped under `[profile.envprofile]`.
#     This proves it is the empty flag, not an inert env var, that produced the
#     top-level recording in step 14.
isolate-home env_profile_only
env ANTITHESIS_PROFILE=envprofile
stdin login_empty_profile.txt
snouty login
file ~/.config/snouty/settings.toml '\[profile\.envprofile\]'
file ~/.config/snouty/settings.toml 'tenant = "atenant"'
file ~/.config/snouty/settings.toml 'repository = "arepo"'

# 16. A blank repository is not recorded by login.The tenant is persisted but `repository`
#     is omitted entirely, so a later command is free to resolve it from ANTITHESIS_REPOSITORY
#     instead of being pinned to a persisted empty value.
isolate-home blank_repository
env ANTITHESIS_PROFILE=""
stdin login_blank_repo.txt
snouty login
file ~/.config/snouty/settings.toml 'tenant = "btenant"'
! file ~/.config/snouty/settings.toml 'repository'
! file ~/.config/snouty/settings.toml '\[profile'
